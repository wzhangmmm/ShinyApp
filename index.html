Stock price downloaded from reliable source

library(quantmod)
dataInput <- to.monthly(getSymbols("SPY", src = "yahoo",
      from = as.Date("01/01/10", format="%m/%d/%y"),
      to = as.Date("01/01/16", format="%m/%d/%y"),
      auto.assign = FALSE))
